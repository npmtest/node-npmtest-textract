  // See https://github.com/dbashford/textract/issues/75 for description of
  // what is happening here
  var pdftotextOptions = options.pdftotextOptions || { layout: 'raw' };

    var fullText;

}

function _compareSlides( a, b ) {
  if ( a.slide < b.slide ) {
    return -1;
  }
  if ( a.slide > b.slide ) {
    return 1;
  }
  return 0;
}
 
function _calculateExtractedText( slideText ) {
  var doc = new Dom().parseFromString( slideText )
    , ps = xpath.select( "//*[local-name()='p']", doc )
    , text = ''
    ;
 
  ps.forEach( function( paragraph ) {
    var ts
      , localText = ''
      ;
 
    paragraph = new Dom().parseFromString( paragraph.toString() );
    ts = xpath.select( "//*[local-name()='t' or local-name()='tab' or local-name()='br']",
      paragraph );
    ts.forEach( function( t ) {
      if ( t.localName === 't' && t.childNodes.length > 0 ) {

          localText += '';
        }
      }

}
 
function extractText( filePath, options, cb ) {
  var slides = [];
 
  yauzl.open( filePath, function( err, zipfile ) {
    if ( err ) {
      util.yauzlError( err, cb );
      return;

    zipfile.on( 'end', function() {
      var slidesText, text;
      if ( slides.length ) {
        slides.sort( _compareSlides );
        slidesText = slides.map( function( slide ) {

  , jschardet = require( 'jschardet' )

function extractText( filePath, options, cb ) {
  fs.readFile( filePath, function( error, data ) {
    var encoding, decoded;
    if ( error ) {
      cb( error, null );
      return;
    }
    try {
      encoding = jschardet.detect( data ).encoding.toLowerCase();
      decoded = iconv.decode( data, encoding );
    } catch ( e ) {
      cb( e );
      return;
    }
    cb( null, decoded );
  });
}
